int global_counter = 0; 
def swap(int ref a, int ref b) -> int {   
    int temp = a;   
    a = b;   
    b = temp;   
    global_counter = global_counter + 1;   
    return 1;   
} 
def main() -> int {   
    int x = 10;   
    int y = 20;   
    swap(x, y);   
    print(x); // Salida: 20   
    print(y); // Salida: 10   
    print(global_counter); // Salida: 1   
    return 0;   
}
